and_or_command   ::= pipeline { ("&&" | "||") pipeline }

pipeline         ::= command { "|" command }

command          ::= simple_command
                   | subshell

subshell         ::= "(" and_or_command ")"

simple_command ::= { redirection } WORD { WORD } { redirection }

redirection      ::= io_number? redirection_op WORD

redirection_op   ::= ">" | ">>" | "<" | "<<"

io_number        ::= DIGIT+
